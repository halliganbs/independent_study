# One Hot Encodes the dataframe 
# creates an np array (number_samples, length_sample, number_values)
def one_hot_encode(df):
  n_samples = len(df['window'])
  X = np.zeros((n_samples, 501, 4))
  for i, seq in enumerate(df['window']):
    for j, nucleotide in enumerate(seq):
      # Possible Nucleotides: A, C, G, T
      if nucleotide == 'A':
        X[i,j,0] = 1
      elif nucleotide == 'T':
        X[i,j,1] = 1
      elif nucleotide == 'C':
        X[i,j,2] = 1
      elif nucleotide == 'G':
        X[i, j, 3] = 1
    
  return X

def build_model(scores, batch_size=32, embedding_dims=50, filters=250, kernel_size=8, 
                hidden_dims=250, epochs=20, dropout_rate=0.2):
  
    model_acc = np.empty([5])
    model_prec = np.empty([5])
    model_recall = np.empty([5])

    index = 0

    for train_index, test_index in kf.split(x, y):
  
        # print('x_train shape:', x_train.shape)
        # print('x_test shape:', x_test.shape)

        print('Iteration: ',index)
        print('Build model...') 
        model = Sequential()

        # Convolution1D, which will learn filters
        # word group filters of size filter_length:
        model.add(Conv1D(filters,
                         kernel_size,
                        padding='valid',
                        activation='relu',
                        strides=1))
        # we use average pooling:
        model.add(GlobalAveragePooling1D())

        # We add a vanilla hidden layer:
        model.add(Dense(hidden_dims))
        model.add(Dropout(dropout_rate))
        model.add(Activation('relu'))

        # We project onto a single unit output layer, and squash it with a sigmoid:
        model.add(Dense(1))
        model.add(Activation('sigmoid'))

        model.compile(loss='binary_crossentropy',
                      optimizer='adam',
                      metrics=['accuracy'])

        # Get the data from the numpy arrays
        x_train, x_test = x[train_index, :, :], x[test_index, :, :]
        y_train, y_test = y[train_index], y[test_index]

        model.fit(x_train, y_train,
                  batch_size=batch_size,
                  epochs=epochs,
                  validation_split=0.0)
  
        # predict
        pred_prob = model.predict(x_test).reshape(-1)
        print(pred_prob.shape, y_test.shape)
        pred_y = np.ones(y_test.shape)
        pred_y[pred_prob < 0.5] = 0.0

        # SKlearn acc, pred, recall
        acc = accuracy_score(y_test, pred_y)
        prec = precision_score(y_test, pred_y)
        recall = recall_score(y_test, pred_y)

        # sanity check
        print('Trial Accuracy: ', acc)
        print('Trial Precision: ', prec)
        print('Trial Recall: ', recall)
        print()

        # add to model values
        model_acc[index] = acc
        model_prec[index] = prec
        model_recall[index] =recall

        # increment index
        index = index +1

    # Get Average of the models
    avg_model_acc = np.mean(model_acc)
    avg_model_prec = np.mean(model_prec)
    avg_model_recall = np.mean(model_recall)
    
    # STD for model
    std_acc = np.std(model_acc)
    std_prec = np.std(model_prec)
    std_recall = np.std(model_recall)
    
    # Add scores to list
    scores.append([avg_model_acc, avg_model_prec, avg_model_recall])
    print('Model Results')
    print('Accuracy: ',avg_model_acc, ' STD: ', std_acc)
    print('Precision: ', avg_model_prec, ' STD: ', std_prec)
    print('Recall: ', avg_model_recall, ' STD: ', std_recall)
    print()
    
    return model_acc, model_prec, model_recall

def get_scores(scores):
    acc = []
    precision = []
    recall = []
    for scr in scores:
        acc.append(scr[0])
        precision.append(scr[1])
        recall.append(scr[2])
        
    acc=np.asarray(acc)
    precision=np.asarray(precision)
    recall=np.asarray(recall)
    
    return acc, precision, recall

def graph_scores(acc, prec, recall, values, type="N/A"):

    plt.plot('x','prec', data=df, color='red', label='Precision')
    plt.plot('x','recall', data=df, color='green', label='Recall')
    plt.xticks(np.arange(min(df['x']), max(df['x'])+1, 1.0))
    plt.legend()
